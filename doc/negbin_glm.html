Simulate data under NB GLM.

N = 1000
P = 10
X = matrix(rnorm(N*P), N, P)
true_conc = 5. 
true_b = rnorm(P)
y = MASS::rnegbin(N, mu = exp(X %*% true_b), theta = true_conc)
df = as.data.frame(X)
df$y = y
